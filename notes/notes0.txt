major = None
minor = None

n = int(input("Digite um número: "))
while n != -1:
    
    if (major is None) or (n > major):
       major = n
    if (minor is None) or (n < minor):
        
       minor = n
    n = int(input("Digite um número: "))
    
print("Maior número é:",major)
print("Menor número é:",minor)
